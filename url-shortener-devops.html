schema.sql
-- URLs table
CREATE TABLE urls (
  id          SERIAL PRIMARY KEY,
  original    TEXT NOT NULL,
  short_code  VARCHAR(10) UNIQUE NOT NULL,
  user_ip     VARCHAR(50),
  created_at  TIMESTAMPTZ DEFAULT NOW(),
  expires_at  TIMESTAMPTZ
);

-- Clicks analytics table
CREATE TABLE clicks (
  id          SERIAL PRIMARY KEY,
  short_code  VARCHAR(10) REFERENCES urls(short_code),
  clicked_at  TIMESTAMPTZ DEFAULT NOW(),
  country     VARCHAR(50),
  device      VARCHAR(50),
  referrer    TEXT
);

routes/url.js
// POST /api/shorten
router.post('/shorten', async (req, res) => {
  const { url, customCode } = req.body;
  const code = customCode || nanoid(6);
  await db.query(
    'INSERT INTO urls(original,short_code) VALUES($1,$2)',
    [url, code]
  );
  res.json({ shortUrl: `https://yourdomain.com/${code}` });
});

// GET /:code — redirect + track
router.get('/:code', async (req, res) => {
  let cached = await redis.get(req.params.code);
  if (!cached) {
    const row = await db.query(
      'SELECT original FROM urls WHERE short_code=$1',
      [req.params.code]
    );
    cached = row.rows[0]?.original;
    await redis.setex(req.params.code, 3600, cached);
  }
  logClick(req.params.code, req); // async, don't await
  res.redirect(301, cached);
});

docker-compose.yml
version: '3.8'
services:
  frontend:
    build: ./frontend
    ports: ["3000:3000"]
    env_file: .env

  backend:
    build: ./backend
    ports: ["5000:5000"]
    depends_on: [postgres, redis]
    env_file: .env

  postgres:
    image: postgres:15
    volumes: [pgdata:/var/lib/postgresql/data]
    environment:
      POSTGRES_DB: urldb
      POSTGRES_PASSWORD: secret

  redis:
    image: redis:7-alpine

  nginx:
    image: nginx:alpine
    volumes: [./nginx.conf:/etc/nginx/nginx.conf]
    ports: ["80:80"]
    depends_on: [frontend, backend]

volumes:
  pgdata:
nginx.conf
server {
  listen 80;

  location /api/ {
    proxy_pass http://backend:5000/;
  }

  location / {
    proxy_pass http://frontend:3000/;
  }
}

Phase 5 — GitHub Actions CI/CD
.github/workflows/deploy.yml
name: Deploy

on:
  push:
    branches: [main]

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3

      - name: Build Docker images
        run: docker compose build

      - name: Push to Docker Hub
        run: |
          echo ${{ secrets.DOCKER_PASSWORD }} | docker login -u ${{ secrets.DOCKER_USERNAME }} --password-stdin
          docker compose push

      - name: SSH deploy to server
        uses: appleboy/ssh-action@master
        with:
          host: ${{ secrets.SERVER_HOST }}
          username: ubuntu
          key: ${{ secrets.SSH_KEY }}
          script: |
            cd ~/url-shortener
            git pull
            docker compose pull
            docker compose up -d --force-recreate

🤖 Vibe Code Prompts
Prompt 1 — Backend scaffold
"Build me a Node.js + Express REST API for a URL shortener. It should have POST /shorten (accepts a URL, generates a 6-char nanoid short code, saves to PostgreSQL), GET /:code (look up in Redis cache first, fallback to DB, redirect 301, log click to clicks table), GET /api/analytics/:code (return total clicks, clicks per day for last 30 days, top 5 countries, device breakdown). Use pg for Postgres and ioredis for Redis. Include error handling and a rate limiter middleware."
Prompt 2 — Frontend dashboard
"Build a Next.js 14 analytics dashboard for a URL shortener. There should be a home page with a form to shorten URLs and copy button. A /dashboard/[code] page showing: total clicks count, an AreaChart of clicks per day (last 30 days), a PieChart of device types, a bar chart of top 5 countries. Use Tailwind CSS for styling, recharts for charts, and fetch data from /api/analytics/:code. Make it dark-themed and clean."
Prompt 3 — Docker + Nginx
"Write a docker-compose.yml that orchestrates: Next.js frontend on port 3000, Express backend on port 5000, PostgreSQL 15 with a named volume, Redis 7 alpine, and Nginx as reverse proxy routing /api/* to backend and /* to frontend. Also write the nginx.conf and a multi-stage Dockerfile for both the frontend and backend for production builds."
Prompt 4 — GitHub Actions CI/CD
"Write a GitHub Actions workflow that triggers on push to main. It should: build and push Docker images to Docker Hub, then SSH into my production server, pull the latest images, and run docker compose up -d --force-recreate. Use these secrets: DOCKER_USERNAME, DOCKER_PASSWORD, SERVER_HOST, SSH_KEY."
